Hook + Component

const orb = useVoiceOrb({
  onQuestionSelect: handleQuestion,
  onInputSubmit: handleInput,
});

<InteractiveVoiceOrb
  ref={orb.orbRef}
  isActive={orb.isOrbActive}
  isListening={orb.isOrbListening}
  isPulsing={orb.orbPulse}
  size="medium"
/>
